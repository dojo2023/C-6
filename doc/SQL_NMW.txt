【IDPWテーブルを作成するSQL文】
DROP TABLE calendars;

CREATE TABLE users (
  u_id VARCHAR(20) PRIMARY KEY,
  password VARCHAR(20) NOT NULL,
  position int default 1
);

CREATE TABLE recipes(
  rec_id int PRIMARY KEY AUTO_INCREMENT,
  r_name VARCHAR(50), 
  time VARCHAR(30),
  image VARCHAR(50),
  wanpan BOOLEAN DEFAULT FALSE,
  save_time BOOLEAN DEFAULT FALSE,
  microwave_oven BOOLEAN DEFAULT FALSE,
  recipe VARCHAR(600),
  cooking_expenses int,
  eating_out_expenses int
)

CREATE TABLE calendars(
  u_id VARCHAR(20),
  rec_id int NOT NULL,
  date DATE,
  c_count int,
  PRIMARY KEY(u_id, rec_id, date)
);
